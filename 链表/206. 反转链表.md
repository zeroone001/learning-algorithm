/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
const reverseList = (head) => {
    let current = head;
    let prev = null;
    let next = head;

    while (current) {
        next = current.next;
        current.next = prev;

        prev = current;

        current = next;
    }
    return prev;

};

var node2 = {val: 1, next: {val:2, next: {val: 3}}};

reverseList(node2);
